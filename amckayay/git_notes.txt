git add - add file to staging area 
git commit - commit added files to staging area, must include message
git init - initialize a new repo
git clone - create a working copy of a remote or local repo
git push origin master - push staged changes to the remote master
git status - check status in local repo
git branch - list all branches
git checkout branchname - switch to branch
git checkout -b branchname - create branch and switch to it
git pull - fetch changes from remote repo and merge with local
git merge branchname - merge changes from another branch into into active branch
